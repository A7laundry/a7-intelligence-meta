## Notas Técnicas

Exemplo de teste para o optimizer:

```python
def test_pause_high_cpa(mock_client):
    """Ad set com CPA > 50 deve ser marcado para pausar."""
    optimizer = CampaignOptimizer()
    optimizer.client = mock_client

    mock_client.list_ad_sets.return_value = [{
        "id": "123", "name": "Test", "status": "ACTIVE", "daily_budget": "5000"
    }]
    mock_client.get_ad_set_insights.return_value = [{
        "impressions": "1000", "clicks": "50", "ctr": "5.0",
        "cpc": "1.0", "spend": "200", "reach": "800",
        "actions": [{"action_type": "lead", "value": "2"}],
        "cost_per_action_type": [{"action_type": "lead", "value": "100"}],
    }]

    result = optimizer.run_optimization(dry_run=True)
    assert any(a["rule"] == "pause_high_cpa" for a in result["actions"])
```
